📋 구글시트 연동 방법
새 스프레드시트도 사용 가능합니다. 기존 연결을 해제하고 새 URL을 등록하면 됩니다.
1
구글 스프레드시트 생성
drive.google.com → 새 스프레드시트 생성
시트 이름을 LOGS로 변경 (A1부터 헤더 자동 생성됨)
2
Apps Script 열기
스프레드시트 상단 메뉴 → 확장 프로그램 → Apps Script
3
Code.gs에 아래 코드 붙여넣기
function doPost(e){
  try{
    var d=JSON.parse(e.postData.contents);
    var ss=SpreadsheetApp.getActiveSpreadsheet();
    var sh=ss.getSheetByName(d.sheet||'LOGS')||ss.insertSheet(d.sheet||'LOGS');
    if(sh.getLastRow()===0){
      sh.appendRow(d.headers||['날짜','현장','업체','장비번호','상태','입력시각']);
    }
    if(d.rows){ d.rows.forEach(function(r){ sh.appendRow(r); }); }
    else if(d.row){ sh.appendRow(d.row); }
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }catch(err){
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e){
  return ContentService.createTextOutput(JSON.stringify({ok:true,msg:'AJ연동 정상'})).setMimeType(ContentService.MimeType.JSON);
}
4
웹앱으로 배포
Apps Script 우측 상단 배포 → 새 배포
유형: 웹앱 선택

액세스 권한: 모든 사용자

URL 복사 → 위에 붙여넣기

⚠️ 새 스프레드시트 연결 시: 위 과정을 새 시트에서 다시 진행하고, URL만 교체하면 됩니다. 기존 앱 데이터는 localStorage에 그대로 유지됩니다.

-- 고소작업대 가동현황 v3.4 — Supabase SQL

-- 1. 테이블 생성
create table if not exists logs (
  id          bigint generated always as identity primary key,
  record_id   text unique not null,
  date        text,        -- YYYY-MM-DD
  site_id     text,
  site_name   text,
  company     text,
  floor       text,
  equip       text,
  recorder    text,
  status      text,        -- start | end | idle
  start_time  text,
  end_time    text,
  used_hours  numeric default 0,
  meter_start text,
  meter_end   text,
  off_reason  text,
  created_at  timestamptz default now()
);

create table if not exists transit (
  id               bigint generated always as identity primary key,
  record_id        text unique not null,
  date             text,
  type             text,    -- 반입 | 반출
  site_id          text,
  site_name        text,
  company          text,
  equip_specs      text,
  aj_equip         text,
  reporter_name    text,
  reporter_phone   text,
  manager_name     text,
  manager_phone    text,
  manager_location text,
  note             text,
  status           text,   -- 대기 | 완료 | 취소
  messages         text,   -- JSON array
  created_at       timestamptz default now()
);

create table if not exists as_requests (
  id             bigint generated always as identity primary key,
  record_id      text unique not null,
  date           text,
  site_id        text,
  site_name      text,
  company        text,
  equip          text,
  location       text,
  fault_type     text,
  description    text,
  reporter_name  text,
  reporter_phone text,
  status         text,     -- 대기 | 처리중 | 완료
  tech_name      text,
  resolved_at    timestamptz,
  resolve_note   text,
  created_at     timestamptz default now()
);

create table if not exists members (
  id        bigint generated always as identity primary key,
  record_id text unique not null,
  name      text,
  company   text,
  site_id   text,
  site_name text,
  phone     text,
  title     text,
  joined_at timestamptz default now()
);

-- 2. 성능 인덱스 (3000대+ 환경 최적화)
-- logs: 날짜·현장·상태 복합 조회용
create index if not exists idx_logs_date       on logs(date);
create index if not exists idx_logs_site_date  on logs(site_id, date);
create index if not exists idx_logs_status     on logs(status);
create index if not exists idx_logs_company    on logs(company);
create index if not exists idx_logs_created    on logs(created_at desc);

-- transit: 현장·날짜·상태 조회용
create index if not exists idx_transit_site    on transit(site_id, date);
create index if not exists idx_transit_status  on transit(status);

-- as_requests: 현장·상태 조회용
create index if not exists idx_as_site         on as_requests(site_id, date);
create index if not exists idx_as_status       on as_requests(status);

-- 3. RLS 비활성화 (내부 전용 — anon key 직접 접근)
alter table logs         disable row level security;
alter table transit      disable row level security;
alter table as_requests  disable row level security;
alter table members      disable row level security;

-- 4. 장비 마스터 테이블 (별도 관리)
create table if not exists equipment (
  id          bigint generated always as identity primary key,
  record_id   text unique not null,
  equip_no    text not null,
  site_id     text,
  site_name   text,
  company     text,
  spec        text,
  model       text,
  transit_id  text,
  status      text default 'active',  -- active | out
  in_date     text,
  out_date    text,
  created_at  timestamptz default now()
);
create index if not exists idx_equip_site    on equipment(site_id, status);
create index if not exists idx_equip_company on equipment(company, status);
create index if not exists idx_equip_no      on equipment(equip_no);
alter table equipment disable row level security;

-- 5. members 테이블 Google 로그인 / 승인 시스템 컬럼 추가
alter table members add column if not exists google_email text;
alter table members add column if not exists status text default 'approved';
alter table members add column if not exists role text default 'sub';
create index if not exists idx_members_google on members(google_email) where google_email is not null;
create index if not exists idx_members_status on members(status);

-- 6. AJ 관리자 테이블 (Supabase에 없을 경우 생성)
create table if not exists aj_members (
  emp_no     text primary key,
  name       text not null,
  phone      text default '',
  pw_hash    text not null,
  aj_type    text default '관리자',
  google_email text,            -- Google 로그인 연동용 (선택)
  created_at timestamptz default now()
);
alter table aj_members disable row level security;
-- 이미 aj_members 테이블이 있는 경우 컬럼 추가
alter table aj_members add column if not exists google_email text;
alter table aj_members add column if not exists status text default 'approved';
create index if not exists idx_aj_google on aj_members(google_email)
  where google_email is not null;
create index if not exists idx_aj_status on aj_members(status);
